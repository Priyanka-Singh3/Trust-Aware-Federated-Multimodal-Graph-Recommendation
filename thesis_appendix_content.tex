% PSEUDOCODE: Federated Training Workflow
% ============================================================================
\begin{algorithm}[htbp]
\caption{Federated Training Workflow for T-FedMMG}
\label{alg:federated_training}
\begin{algorithmic}[1]
\REQUIRE Number of clients $K$, local epochs $E$, learning rate $\eta$, \\
\quad batch size $B$, privacy budget $\epsilon$
\ENSURE Global model parameters $\theta^{(T)}$
\STATE Initialize global model $\theta^{(0)}$
\STATE Partition dataset $\mathcal{D}$ into $K$ subsets $\{\mathcal{D}_1, \ldots, \mathcal{D}_K\}$
\FOR{each communication round $t = 1, \ldots, T$}
    \STATE $S_t \leftarrow$ Select subset of clients (if not full participation)
    \FOR{each client $k \in S_t$ \textbf{in parallel}}
        \STATE $\theta_k^{(t,0)} \leftarrow \theta^{(t-1)}$ \quad \COMMENT{Download global model}
        \FOR{each local epoch $e = 1, \ldots, E$}
            \FOR{each batch $\mathcal{B} \subset \mathcal{D}_k$}
                \STATE Sample negative items $\mathcal{N}(u)$ for each user $u$
                \STATE Compute BPR loss: $\mathcal{L} = -\sum_{(u,i,j)} \ln \sigma(\hat{r}_{ui} - \hat{r}_{uj})$
                \STATE Compute gradients: $g_k \leftarrow \nabla \mathcal{L}(\theta_k^{(t,e)})$
                \STATE Clip gradients: $\tilde{g}_k \leftarrow g_k / \max(1, \frac{\|g_k\|_2}{C})$
                \STATE Add noise: $\hat{g}_k \leftarrow \tilde{g}_k + \mathcal{N}(0, \sigma^2 C^2 I)$
                \STATE Update: $\theta_k^{(t,e+1)} \leftarrow \theta_k^{(t,e)} - \eta \cdot \hat{g}_k$
            \ENDFOR
        \ENDFOR
        \STATE Upload $\theta_k^{(t,E)}$ to server
    \ENDFOR
    \STATE \COMMENT{Server aggregation with trust weighting}
    \FOR{each client $k$}
        \STATE Compute trust score: $\tau_k \leftarrow \text{ComputeTrust}(k)$
        \STATE Compute update quality: $q_k \leftarrow \text{CosineSimilarity}(\theta_k - \theta^{(t-1)}, \bar{\Delta})$
    \ENDFOR
    \STATE Aggregate: $\theta^{(t)} \leftarrow \sum_{k} \frac{\tau_k \cdot n_k}{\sum_j \tau_j \cdot n_j} \cdot \theta_k^{(t,E)}$
    \STATE Broadcast $\theta^{(t)}$ to all clients
    \STATE Evaluate global model on validation set
    \IF{convergence criterion met}
        \STATE \textbf{break}
    \ENDIF
\ENDFOR
\RETURN $\theta^{(T)}$
\end{algorithmic}
\end{algorithm}

\section{Dataset Description}
\label{app:dataset}

\subsection{Yelp Dataset Overview}

The Yelp dataset used in this study consists of business reviews and metadata from the Yelp Open Dataset. We focus on the metropolitan area of Philadelphia, PA, USA, which provides a diverse set of businesses and user interactions suitable for recommendation system research.

\subsection{Data Preprocessing}

\textbf{User Filtering:} Users with fewer than 1 interaction and more than 100 interactions were excluded to ensure sufficient data for training while removing potential spammers.

\textbf{Business Filtering:} Businesses with fewer than 3 reviews were filtered out to ensure adequate representation in the feature extraction phase.

\textbf{Train-Test Split:} We employ a \textbf{leave-one-out} evaluation protocol:
\begin{itemize}
    \item For each user, the most recent interaction is held out for testing
    \item The second most recent interaction is used for validation
    \item Remaining interactions are used for training (80\%)
\end{itemize}

\textbf{Negative Sampling:} For evaluation, we use a sampled protocol with 1 positive item and 99 randomly sampled negative items per user, following standard recommendation system evaluation practices.

\subsection{Feature Extraction}

\textbf{Text Features:} Business categories and review text are combined and processed using TF-IDF vectorization with a maximum of 1,000 features, including unigrams and bigrams. Stop words are removed and text is lowercased.

\textbf{Image Features:} Business photos are processed using a pre-trained ResNet-18 CNN model (trained on ImageNet) to extract 512-dimensional visual embeddings. The CNN backbone remains frozen during training.

\section{Trust Computation Workflow}
\label{app:trust_workflow}

\subsection{Trust Score Components}

The trust score $T(u,i)$ for a user-item interaction is computed as a weighted combination of four components:

\begin{equation}
T(u,i) = \alpha \cdot T_{\text{consistency}} + \beta \cdot T_{\text{popularity}} + \gamma \cdot T_{\text{recency}} + \delta \cdot T_{\text{activity}}
\end{equation}

where $\alpha + \beta + \gamma + \delta = 1$ (typically $\alpha=0.3, \beta=0.25, \gamma=0.25, \delta=0.2$).

\subsection{Component Definitions}

\begin{enumerate}
    \item \textbf{Rating Consistency ($T_{\text{consistency}}$):} Measures how consistent a user's ratings are with their historical pattern.
    \begin{equation}
    T_{\text{consistency}}(u,i) = 1 - \frac{|r_{ui} - \bar{r}_u|}{\max(|r_{ui} - \bar{r}_u|)}
    \end{equation}
    where $\bar{r}_u$ is the user's average rating.
    
    \item \textbf{Item Popularity ($T_{\text{popularity}}$):} Reflects the overall trustworthiness of the item based on community feedback.
    \begin{equation}
    T_{\text{popularity}}(i) = \frac{\text{count}(r_i > 3)}{\text{total\_ratings}(i)}
    \end{equation}
    
    \item \textbf{Recency ($T_{\text{recency}}$):} More recent interactions are weighted higher.
    \begin{equation}
    T_{\text{recency}}(u,i) = \exp(-\lambda \cdot (t_{\text{current}} - t_{ui}))
    \end{equation}
    where $\lambda$ is a decay factor (typically 0.1 per week).
    
    \item \textbf{User Activity ($T_{\text{activity}}$):} Active users with diverse interactions are considered more trustworthy.
    \begin{equation}
    T_{\text{activity}}(u) = \min\left(1, \frac{\text{interactions}(u)}{\text{threshold}}\right)
    \end{equation}
    where threshold is typically set to 10 interactions.
\end{enumerate}

\subsection{Trust Computation Algorithm}

\begin{algorithm}[htbp]
\caption{Trust Score Computation}
\label{alg:trust_computation}
\begin{algorithmic}[1]
\REQUIRE User $u$, Item $i$, Rating $r_{ui}$, Timestamp $t_{ui}$, User history $\mathcal{H}_u$
\ENSURE Trust score $T(u,i) \in [0,1]$
\STATE $T_{\text{consistency}} \leftarrow 1 - |r_{ui} - \text{mean}(\mathcal{H}_u)| / 4$ \COMMENT{Normalized by max rating diff}
\STATE $T_{\text{popularity}} \leftarrow \text{positive\_ratio}(i)$ \COMMENT{Fraction of ratings > 3
\STATE $T_{\text{recency}} \leftarrow \exp(-0.1 \cdot (t_{\text{now}} - t_{ui}))$
\STATE $T_{\text{activity}} \leftarrow \min(1, |\mathcal{H}_u| / 10)$
\STATE $T(u,i) \leftarrow 0.3 \cdot T_{\text{consistency}} + 0.25 \cdot T_{\text{popularity}}$
\STATE \quad $+\, 0.25 \cdot T_{\text{recency}} + 0.2 \cdot T_{\text{activity}}$
\RETURN $T(u,i)$
\end{algorithmic}
\end{algorithm}

\subsection{Trust-Aware Recommendation}

The final recommendation score combines the predicted rating with the trust score:

\begin{equation}
S(u,i) = w_1 \cdot \hat{r}_{ui} + w_2 \cdot T(u,i)
\end{equation}

where $w_1$ and $w_2$ are weights (typically $w_1=0.6, w_2=0.4$) balancing the predicted rating against the trustworthiness of the interaction.
